 package org.example;
 
 import java.util.List;
 
 public class ListService {
     public ListService() {
     }
 
     public double getAverage(List<Integer> list) {
         return list.stream()
                 .mapToInt(s -> s)
                 .average().orElse(0);
     }
 
     public String compareAverage(List<Integer> list1, List<Integer> list2) {
         if (getAverage(list1) > getAverage(list2)) return "Первый список имеет большее среднее значение";
         else if (getAverage(list1) < getAverage(list2)) return "Второй список имеет большее среднее " +
                 "значение";
         else return "Средние значения равны";
     }
 }
